@event_handler('material_entrada')
def criar_conta_pagar_entrada_material(data: dict, admin_id: int):
    """Handler: Criar conta a pagar quando material entra no estoque"""
    try:
        from models import db, AlmoxarifadoMovimento, ContaPagar, Fornecedor
        from datetime import datetime, timedelta
        from decimal import Decimal
        
        movimento_id = data.get('movimento_id')
        
        if not movimento_id:
            logger.warning(f"âš ï¸ movimento_id nÃ£o fornecido no evento material_entrada")
            return
        
        # Buscar movimento
        movimento = AlmoxarifadoMovimento.query.filter_by(
            id=movimento_id,
            admin_id=admin_id
        ).first()
        
        if not movimento:
            logger.error(f"âŒ Movimento {movimento_id} nÃ£o encontrado")
            return
        
        # Validar fornecedor
        if not movimento.fornecedor_id:
            logger.info(f"â­ï¸ Conta a pagar nÃ£o criada: movimento sem fornecedor")
            return
        
        # Calcular valor total
        quantidade = float(movimento.quantidade or 0)
        valor_unitario = float(movimento.valor_unitario or 0)
        valor_total = Decimal(str(quantidade * valor_unitario))
        
        if valor_total <= 0:
            logger.info(f"â­ï¸ Conta a pagar nÃ£o criada: valor zerado")
            return
        
        # Verificar se jÃ¡ existe conta para esta nota fiscal
        if movimento.nota_fiscal:
            conta_existente = ContaPagar.query.filter_by(
                admin_id=admin_id,
                fornecedor_id=movimento.fornecedor_id,
                numero_documento=movimento.nota_fiscal
            ).first()
            
            if conta_existente:
                logger.info(f"â­ï¸ Conta a pagar jÃ¡ existe para NF {movimento.nota_fiscal}")
                return
        
        # Buscar fornecedor
        fornecedor = Fornecedor.query.get(movimento.fornecedor_id)
        
        # Criar conta a pagar
        conta = ContaPagar(
            admin_id=admin_id,
            fornecedor_id=movimento.fornecedor_id,
            numero_documento=movimento.nota_fiscal or f"MOV-{movimento_id}",
            descricao=f"Compra de materiais - {movimento.item.nome if movimento.item else 'Material'}",
            valor_original=valor_total,
            valor_pago=Decimal('0'),
            saldo=valor_total,
            data_emissao=movimento.data_movimento.date() if movimento.data_movimento else datetime.now().date(),
            data_vencimento=(movimento.data_movimento + timedelta(days=30)).date() if movimento.data_movimento else (datetime.now() + timedelta(days=30)).date(),
            status='PENDENTE',
            conta_contabil_codigo='2.1.01.001'  # Fornecedores a Pagar
        )
        
        db.session.add(conta)
        db.session.commit()
        
        logger.info(f"âœ… Conta a pagar criada: {conta.id} - R$ {valor_total} - Fornecedor: {fornecedor.razao_social if fornecedor else 'N/A'}")
        
    except Exception as e:
        db.session.rollback()
        logger.error(f"âŒ Erro ao criar conta a pagar: {str(e)}")
