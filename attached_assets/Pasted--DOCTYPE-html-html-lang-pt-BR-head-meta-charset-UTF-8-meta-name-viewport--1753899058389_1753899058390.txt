        // Função para alternar entre dashboards
        function showDashboard(dashboardId) {
            // Esconder todos os dashboards
            document.querySelectorAll('.dashboard-tab').forEach(tab => {
                tab.classList.remove('active');
            });
            
            // Remover classe active de todos os links
            document.querySelectorAll('.nav-link').forEach(link => {
                link.classList.remove('active');
            });
            
            // Mostrar dashboard selecionado
            document.getElementById(dashboardId).classList.add('active');
            
            // Adicionar classe active ao link clicado
            event.target.classList.add('active');
            
            // Inicializar gráficos do dashboard ativo
            initializeCharts(dashboardId);
        }
